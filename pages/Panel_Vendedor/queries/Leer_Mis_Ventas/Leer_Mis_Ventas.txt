SELECT 
  o.fecha_orden,
  cm.nombre_item,
  det.cantidad,
  -- AQUÍ ESTÁ EL BLINDAJE: Leemos el precio original del inventario (ir)
  ir.precio_base AS precio_original,
  -- Calculamos el total usando el precio del proveedor
  (det.cantidad * ir.precio_base) AS total_ganado,
  det.estado_pago
FROM detalle_ordenes det
JOIN ordenes o ON det.id_orden = o.id_orden
JOIN inventario_real ir ON det.id_lote = ir.id_lote
JOIN catalogo_maestro cm ON ir.item_id = cm.item_id
WHERE (
  ({{appsmith.store.piloto_rol}} = 'admin')
  OR
  (ir.proveedor_id::text = {{appsmith.store.piloto_id.toString()}})
)
AND o.estado_logistico = 'Entregado'
ORDER BY o.fecha_orden DESC;
